\newarray\IdArr
\newcounter{IdArrSize}
\setcounter{IdArrSize}{0}

\makeatletter

\newcommand{\defn}[3]{
  \stepcounter{IdArrSize}
  \IdArr(\value{IdArrSize})={#1}

  \begin{definition}
    \label{def:#1}
    \textbf{#1}\par
    #2\par
    \ifthenelse{\equal{#3}{}}{{\color{red} No significance?}}{\textbf{Significance}:#3}\par
  \end{definition}

  \@ifnextchar\bgroup{
    \textbf{Dependencies:} \parsenextparent
  }{
    \textbf{Axiom}
  }
}

\newcommand{\parsenextparent}[1]{
  #1 [\ref{def:#1}]
  \@ifnextchar\bgroup{
    \parsenextparent
  }{}
}

\makeatother

\endinput
