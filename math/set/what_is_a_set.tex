\section{what is a set?}

\begin{definition}
  \label{def:set}

  \textbf{A set is a collection of things (also called elements) in which
  \begin{enumerate}
    \item there is no information of order
    \item no two things picked from the set without replacement, shall be equal
  \end{enumerate}
  }

  \defusecase{To represent collection of differently colored balls in an urn, etc...}
\end{definition}

\begin{definition}
  \label{def:ordered_set}

  \textbf{An ordered set is a collection of elements in which
    \begin{enumerate}
    \item there is information of order
    \item no two things picked from the set without replacement, shall be equal
  \end{enumerate}
  }

  \defusecase{To represent collection of real numbers from 0 to 1, etc...}
\end{definition}

\begin{definition}
  \label{def:set_cardinality}

  \textbf{The cardinality of a set [\ref{def:set}] or an ordered set [\ref{def:ordered_set}] is the number of elements in it.}

  \defusecase{A name for count of elements in the collection. Don't really know why they didn't call it `size' or `length'}
\end{definition}
