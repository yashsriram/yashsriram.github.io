\chapter{what is a set?}

\defn{Set}{
  A set is a collection of things (also called elements) in which no two things picked without replacement from the collection are equal.
}{
  To represent collection of differently colored balls in an urn, etc...
}

\defn{OrderedSet}{
  A set in which for any two elements picked without replacement, one element is higher than the other
}{
  To represent collection of etc...
}

\defn{SetCardinality}{
  The cardinality of a set or an ordered set is the number of elements in it.
}{
  A name for count of elements in the collection. Don't really know why they didn't call it `size' or `length'
}{Set}{OrderedSet}

\chapter{continuous vs contiguous ordered sets}

\defn{ContinuousSet}{
  An ordered set which has no discrete separation b/w any two of its elements is called a continuous set.
}{
  To represent sets like the set of all wavelengths in the visible spectrum of light
}{OrderedSet}

\defn{ContiguousSet}{
  An ordered set which has a discrete separation b/w any two of its elements is called a contiguous set.
}{
  To represent sets like the ordered set of 7 colors [Violet, Indigo, Blue, Green, Yellow, Orange, Red]
}{OrderedSet}
